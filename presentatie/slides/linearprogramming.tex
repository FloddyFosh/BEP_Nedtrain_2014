\begin{frame}\frametitle{Linear Programming}
    \begin{definitie}
        \begin{align}
            \text{max:}& \quad \sum_{t \in T} (t^+ - t^-) & \nonumber \\
            \text{met voorwaarden:} & \quad 0 \leq t^+ - t^- & \forall t \in T \nonumber \\
                                    & \quad t^+ - t^- \leq c & \forall (t^+ - t^{'-} \leq c) \in C \nonumber
        \end{align}
    \end{definitie}
\end{frame}
